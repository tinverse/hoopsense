import cv2
import json
from ultralytics import YOLO

def extract_game_dna(video_path, output_json_path):
    print(f"Loading AI Models for {video_path}...")

    # 1. Load YOLO (The 'Perceiver').
    # 'yolov8n.pt' is the nano model. It's fast enough for a laptop GPU.
    model = YOLO('yolov8n.pt')

    # 2. Open the video file
    cap = cv2.VideoCapture(video_path)

    # This will hold our "Player DNA" for the entire game
    game_ledger = []
    frame_count = 0

    print("Starting frame-by-frame inference. (This will take a while...)")

    # 3. The Frame Loop
    while cap.isOpened():
        success, frame = cap.read()
        if not success:
            break

        frame_count += 1

        # Run YOLO with BoT-SORT Tracking enabled.
        # 'persist=True' tells the AI to build that progressively confident memory bank.
        # classes=[0, 32] tells it to ONLY look for humans (0) and sports balls (32).
        results = model.track(frame, persist=True, classes=[0, 32], tracker="botsort.yaml", verbose=False)

        # Extract the data for this specific frame
        frame_data = {
            "frame_id": frame_count,
            "timestamp_sec": frame_count / cap.get(cv2.CAP_PROP_FPS),
            "entities": []
        }

        # 4. Parse the AI's output
        if results[0].boxes.id is not None:
            boxes = results[0].boxes.xywh.cpu().numpy() # [x_center, y_center, width, height]
            track_ids = results[0].boxes.id.int().cpu().numpy()
            class_ids = results[0].boxes.cls.int().cpu().numpy()

            for box, track_id, class_id in zip(boxes, track_ids, class_ids):
                entity_type = "player" if class_id == 0 else "ball"

                frame_data["entities"].append({
                    "id": int(track_id),
                    "type": entity_type,
                    "position": {"x": float(box[0]), "y": float(box[1])},
                    "dimensions": {"w": float(box[2]), "h": float(box[3])}
                })

        game_ledger.append(frame_data)

        # Optional: Print progress every 100 frames
        if frame_count % 100 == 0:
            print(f"Processed {frame_count} frames...")

    cap.release()

    # 5. Save the "Game DNA" to a file
    with open(output_json_path, 'w') as f:
        json.dump(game_ledger, f, indent=4)

    print(f"Success! Game DNA saved to {output_json_path}")

# --- Run the pipeline ---
if __name__ == "__main__":
    VIDEO_FILE = "my_sample_game.mp4" # Replace with your video file
    OUTPUT_FILE = "game_dna_output.json"

    extract_game_dna(VIDEO_FILE, OUTPUT_FILE)

# --- HELPER FUNCTIONS FOR IDENTITY FUSION ---



def get_team_color(crop_img):

    if crop_img is None or crop_img.size == 0:

        return "unknown"

    h, w = crop_img.shape[:2]

    center = crop_img[int(h*0.2):int(h*0.8), int(w*0.2):int(w*0.8)]

    if center.size == 0: center = crop_img

    hsv = cv2.cvtColor(center, cv2.COLOR_BGR2HSV)

    mean_v = np.mean(hsv[:,:,2])

    return "light" if mean_v > 120 else "dark"



def perform_ocr_voting(crops, reader):

    votes = []

    for crop in crops:

        if crop.shape[0] < 20 or crop.shape[1] < 20: continue

        results = reader.readtext(crop)

        for (bbox, text, prob) in results:

            digits = re.sub(r'[^0-9]', '', text)

            if 1 <= len(digits) <= 2 and prob >= 0.6:

                votes.append(digits.zfill(2))

    if not votes: return None

    return Counter(votes).most_common(1)[0][0]



def resolve_player_identity(track_id, crops, reader):

    color_votes = [get_team_color(crop) for crop in crops]

    final_team = Counter(color_votes).most_common(1)[0][0]

    final_number = perform_ocr_voting(crops, reader)

    return {"team": final_team, "jersey_number": final_number}



# --- CHECKPOINT LOGIC ---



def save_checkpoint(checkpoint_path, frame_count, resolved_identities):

    with open(checkpoint_path, 'w') as f:

        json.dump({

            "last_frame": frame_count,

            "identities": resolved_identities

        }, f)



def load_checkpoint(checkpoint_path):

    if os.path.exists(checkpoint_path):

        with open(checkpoint_path, 'r') as f:

            data = json.load(f)

            return data["last_frame"], data["identities"]

    return 0, {}

def extract_game_dna(video_path, drive_folder="HoopSense_Data"):

    # Ensure the video file exists before starting

    if not os.path.exists(video_path):

        print(f"Error: Video file not found at {video_path}")

        print("Please check the path in your Google Drive.")

        return



    output_dir = setup_drive_paths(drive_folder)



    # Use video filename for specific output files to avoid overwriting different games

    video_filename = os.path.splitext(os.path.basename(video_path))[0]

    output_jsonl = os.path.join(output_dir, f"{video_filename}_dna.jsonl")

    checkpoint_path = os.path.join(output_dir, f"{video_filename}_checkpoint.json")



    print(f"Loading AI Models... Device: {'GPU' if torch.cuda.is_available() else 'CPU'}")

    model = YOLO('yolov8n.pt')

    ocr_reader = easyocr.Reader(['en'], gpu=torch.cuda.is_available())



    cap = cv2.VideoCapture(video_path)

    raw_fps = cap.get(cv2.CAP_PROP_FPS)

    fps = raw_fps if raw_fps > 0 else 30.0



    # Resume from checkpoint

    start_frame, resolved_identities = load_checkpoint(checkpoint_path)

    if start_frame > 0:

        print(f"Resuming {video_filename} from frame {start_frame}...")

        cap.set(cv2.CAP_PROP_POS_FRAMES, start_frame)



    track_history = {}

    track_last_seen = {}

    BUFFER_SIZE = 30

    MAX_TRACK_AGE = 60

    frame_count = start_frame



    # Use 'a' (append) mode to preserve existing data on resume

    mode = 'a' if start_frame > 0 else 'w'



    with open(output_jsonl, mode) as f_out:

        while cap.isOpened():

            success, frame = cap.read()

            if not success: break



            frame_count += 1

            t_ms = int((frame_count / fps) * 1000)



            # Expire old track buffers

            active_tracks = list(track_history.keys())

            for tid in active_tracks:

                if frame_count - track_last_seen.get(tid, 0) > MAX_TRACK_AGE:

                    del track_history[tid]

                    del track_last_seen[tid]



            results = model.track(frame, persist=True, classes=[0, 32], tracker="botsort.yaml", verbose=False)



            if results[0].boxes.id is not None:

                boxes = results[0].boxes.xywh.cpu().numpy()

                xyxys = results[0].boxes.xyxy.cpu().numpy()

                track_ids = results[0].boxes.id.int().cpu().numpy()

                class_ids = results[0].boxes.cls.int().cpu().numpy()

                confs = results[0].boxes.conf.cpu().numpy()



                for box, xyxy, track_id, class_id, conf in zip(boxes, xyxys, track_ids, class_ids, confs):

                    entity_type = "player" if class_id == 0 else "ball"

                    track_id = int(track_id)



                    if entity_type == "player":

                        track_last_seen[track_id] = frame_count

                        if track_id not in resolved_identities:

                            if track_id not in track_history:

                                track_history[track_id] = deque(maxlen=BUFFER_SIZE)

                            x1, y1, x2, y2 = map(int, xyxy)

                            torso_crop = frame[max(0, y1):max(0, y1 + int((y2-y1)*0.6)), max(0, x1):min(frame.shape[1], x2)]

                            if frame_count % 2 == 0:

                                track_history[track_id].append(torso_crop)



                            if len(track_history[track_id]) == BUFFER_SIZE:

                                identity = resolve_player_identity(track_id, list(track_history[track_id]), ocr_reader)

                                resolved_identities[track_id] = identity

                                print(f"[Identity Locked] Track {track_id} -> {identity['jersey_number']}")

                                del track_history[track_id]



                    current_identity = resolved_identities.get(track_id, {})

                    row = {

                        "kind": entity_type, "track_id": track_id, "t_ms": t_ms,

                        "x": round(float(box[0]), 2), "y": round(float(box[1]), 2),

                        "w": round(float(box[2]), 2), "h": round(float(box[3]), 2),

                        "confidence_bps": int(conf * 10000),

                        "actor_jersey_number": current_identity.get("jersey_number"),

                        "team_color": current_identity.get("team")

                    }

                    f_out.write(json.dumps(row) + "\n")



            # Periodic Checkpoint

            if frame_count % 500 == 0:

                save_checkpoint(checkpoint_path, frame_count, resolved_identities)

                print(f"Checkpoint saved at frame {frame_count}...")



    # Final cleanup

    save_checkpoint(checkpoint_path, frame_count, resolved_identities)

    cap.release()

    print(f"Success! Game DNA saved for {video_filename} in: {output_jsonl}")

In your specific output folder, it is named [video_filename]_checkpoint.json.

What it contains:

identities: The "Memory Bank" of all resolved players. This stores the mapping of track_id to their specific jersey_number and team_color.

Why it exists:
Resume Capability: If Google Colab disconnects or your runtime stops, the script looks for this file. If it finds it, the AI doesn't have to start from Frame 0. It jumps directly to the last saved frame and picks up where it left off.

Memory Efficiency: By saving the "locked" identities here, the script doesn't have to spend GPU time re-running OCR on players it has already identified earlier in the game.

Persistence: It ensures that if you process half of a video today and half tomorrow, the data remains consistent across sessions.
